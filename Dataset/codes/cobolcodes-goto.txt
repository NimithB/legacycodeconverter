IDENTIFICATION DIVISION.
PROGRAM-ID. HELLO.
DATA DIVISION.
   WORKING-STORAGE SECTION.
   01 WS-NUM1 PIC 9(9).
   01 WS-NUM2 PIC 9(9).
PROCEDURE DIVISION.
   A000-FIRST-PARA.
   MOVE 25 TO WS-NUM1.
   MOVE 15 TO WS-NUM2.
   IF WS-NUM1 > WS-NUM2
      GO TO GREATER
   ELSE
      GO TO NOT-GREATER
   END-IF.
   STOP RUN.
   GREATER.
      DISPLAY 'WS-NUM1 is greater than WS-NUM2'
      GO TO END-PROGRAM.
   NOT-GREATER.
      DISPLAY 'WS-NUM1 is not greater than WS-NUM2'
      GO TO END-PROGRAM.
   END-PROGRAM.
      STOP RUN.

      
      IDENTIFICATION DIVISION.
PROGRAM-ID. HELLO2.
DATA DIVISION.
   WORKING-STORAGE SECTION.
   01 WS-NUM1 PIC 9(9).
   01 WS-NUM2 PIC 9(9).
   01 WS-DIFF PIC 9(9) VALUE ZEROS.
   01 WS-COMPUTE PIC 99 VALUE ZEROS.
PROCEDURE DIVISION.
   A000-FIRST-PARA.
   MOVE 25 TO WS-NUM1.
   MOVE 15 TO WS-NUM2.
   COMPUTE WS-DIFF = WS-NUM1 - WS-NUM2
   IF WS-DIFF > 0
      MOVE 1 TO WS-COMPUTE
   ELSE IF WS-DIFF < 0
      MOVE 2 TO WS-COMPUTE
   END-IF.
   GO TO COMPUTED-GOTO-TABLE (WS-COMPUTE).
   STOP RUN.
   COMPUTED-GOTO-TABLE.
   01 IS GREATER THRU GREATER-END.
      DISPLAY 'WS-NUM1 is greater than WS-NUM2'
      GO TO END-PROGRAM.
   GREATER-END.
   01 IS NOT-GREATER THRU NOT-GREATER-END.
      DISPLAY 'WS-NUM1 is not greater than WS-NUM2'
      GO TO END-PROGRAM.
   NOT-GREATER-END.
   END-PROGRAM.
      STOP RUN.

      IDENTIFICATION DIVISION.
PROGRAM-ID. HELLO3.
DATA DIVISION.
   WORKING-STORAGE SECTION.
   01 WS-NUM1 PIC 9(9).
   01 WS-NUM2 PIC 9(9).
   01 WS-NUM3 PIC 9(9).
PROCEDURE DIVISION.
   A000-FIRST-PARA.
   MOVE 25 TO WS-NUM1.
   MOVE 15 TO WS-NUM2.
   MOVE 30 TO WS-NUM3.
   IF WS-NUM1 > WS-NUM2
      GO TO GREATER-THAN-NUM2
   ELSE IF WS-NUM1 > WS-NUM3
      GO TO GREATER-THAN-NUM3
   ELSE
      GO TO NOT-GREATER
   END-IF.
   STOP RUN.
   GREATER-THAN-NUM2.
      DISPLAY 'WS-NUM1 is greater than WS-NUM2'
      GO TO END-PROGRAM.
   GREATER-THAN-NUM3.
      DISPLAY 'WS-NUM1 is greater than WS-NUM3'
      GO TO END-PROGRAM.
   NOT-GREATER.
      DISPLAY 'WS-NUM1 is not greater than WS-NUM2 or WS-NUM3'
      GO TO END-PROGRAM.
   END-PROGRAM.
      STOP RUN.

      
      IDENTIFICATION DIVISION.
PROGRAM-ID. HELLO4.
DATA DIVISION.
   WORKING-STORAGE SECTION.
   01 WS-NUM PIC 9(9) VALUE ZEROS.
   01 WS-COUNTER PIC 9(4) VALUE 1.
PROCEDURE DIVISION.
   A000-FIRST-PARA.
   PERFORM UNTIL WS-COUNTER > 10
      DISPLAY 'Enter a number: '
      ACCEPT WS-NUM
      IF WS-NUM > 50
         GO TO GREATER-THAN-50
      ELSE
         GO TO NOT-GREATER-THAN-50
      END-IF.
      ADD 1 TO WS-COUNTER.
   END-PERFORM.
   STOP RUN.
   GREATER-THAN-50.
      DISPLAY 'The number is greater than 50'
      GO TO END-PROGRAM.
   NOT-GREATER-THAN-50.
      DISPLAY 'The number is not greater than 50'
      GO TO END-PROGRAM.
   END-PROGRAM.
      STOP RUN.
